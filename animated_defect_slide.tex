% Overlay 3: DFT across 5 charge states
\only<3-4>{
  \begin{tcolorbox}[colback=green!5, colframe=green!60!black,
    title={\small DFT for Defect Energetics}, fonttitle=\bfseries,
    boxrule=0.5pt, arc=2mm, left=2pt, right=2pt, top=2pt, bottom=2pt]
  \small
  Formation energy for charge state $q$: \\[2pt]
  $E_f(q) = E_{\text{def}}^q - E_{\text{bulk}} + \sum n_i\mu_i + q\,E_F$ \\[4pt]
  Must compute \textbf{5 charge states}: \\
  $q \in \{+2, +1, 0, -1, -2\}$ \\[2pt]
  $\Rightarrow$ Charge transition levels (CTLs) \\
  $\Rightarrow$ Identify deep vs.\ shallow defects
  \end{tcolorbox}
}

% Overlay 5: DFT is expensive
\only<5>{
  \begin{tcolorbox}[colback=orange!8, colframe=orange!80!black,
    title={\small DFT Cost Bottleneck}, fonttitle=\bfseries,
    boxrule=0.5pt, arc=2mm, left=2pt, right=2pt, top=2pt, bottom=2pt]
  \small
  Per defect configuration (HSE06): \\[2pt]
  \textbf{512 cores $\times$ 8 h} = 4,096 core-h \\[3pt]
  $\times$ 5 charge states = \textbf{20,480 core-h} \\
  $\times$ 10 symmetry configs = \textbf{204,800 core-h} \\[4pt]
  \textcolor{red!70!black}{$\Rightarrow$ Millions of defects are impractical!}
  \end{tcolorbox}
